def pdist(array, point):
    x, y  = array[:,0], array[:,1]
    return np.sqrt( (x-point[0])**2 + (y-point[1])**2 )

def weber(ta, td, p1, imax=30, tol=0.0000001):
    XX, YY = p1[0], p1[1]
    gap = 9999999999.0
    dd = (pdist(td,p1)) + tol
    k = 0
    
    obj = 0
    for i in range(len(ta)):
        obj += ta[i] * np.sqrt( (td[i,0] - XX)**2 + (td[i,1] - YY)**2 )
    
    while ((k<imax)&(gap>tol)):
        k += 1
        temp = (ta/dd).sum()
        xnew = ((ta*td[:,0])/dd).sum()
        ynew = ((ta*td[:,1])/dd).sum()
        p1 = [xnew / temp, ynew / temp]
        dd = pdist(td,p1) + tol
#         newobj = round(sum(ta*dd), 3) 
# the 3 decimal points resoution as in R will make the function find the solution within 16 iterations.
        newobj = sum(ta*dd)
        gap = obj - newobj
        obj = newobj
    print('the optimal location for the one facility is ', str(p1), ' and it took ', k, ' iterations to find the solution')
    return (p1, k)

for face in faces:
    p1, _ = nearest_points(face, regionalWeber)
    p2, __ = nearest_points(face.boundary, regionalWeber)
    
    faceWebers.append(p1) if p1.distance(regionalWeber) < p2.distance(regionalWeber) else faceWebers.append(p2)
