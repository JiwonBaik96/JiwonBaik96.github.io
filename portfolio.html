On the importance of sentence length

Now listen. I vary the sentence length, and I create music. Music. The writing sings. It has a pleasent rhythm, a lilt, a harmony. I use short sentences. And I use sentences of medium length. And sometimes when I am certain the reader is rested, I will engage him with a sentence of considerable length, a sentence that burns with energy and builds with all the impetus of a crescendo, the roll of the drums, the crash of the cymbals – sounds that say listen to this, it is important.

- Gary Provost (100 Ways to Improve Your Writing, 1985)

import ipywidgets as widgets
from IPython.display import display, HTML

javascript_functions = {False: "hide()", True: "show()"}
button_descriptions  = {False: "Show code", True: "Hide code"}


def toggle_code(state):

    """
    Toggles the JavaScript show()/hide() function on the div.input element.
    """

    output_string = "<script>$(\"div.input\").{}</script>"
    output_args   = (javascript_functions[state],)
    output        = output_string.format(*output_args)

    display(HTML(output))


def button_action(value):

    """
    Calls the toggle_code function and updates the button description.
    """

    state = value.new

    toggle_code(state)

    value.owner.description = button_descriptions[state]


state = False
toggle_code(state)

button = widgets.ToggleButton(state, description = button_descriptions[state])
button.observe(button_action, "value")

display(button)


import numpy as np
import pandas as pd 
import time

import matplotlib.pyplot as plt
from shapely.geometry import Point, LineString
from shapely.ops import unary_union, polygonize
from shapely.ops import nearest_points

from mpl_toolkits.mplot3d import Axes3D 
import matplotlib.pyplot as plt
from matplotlib import cm

import altair as alt


def pdist(array, point):
    x, y  = array[:,0], array[:,1]
    return np.sqrt( (x-point[0])**2 + (y-point[1])**2 )

def weber(ta, td, p1, imax=30, tol=0.0000001):
    XX, YY = p1[0], p1[1]
    gap = 9999999999.0
    dd = (pdist(td,p1)) + tol
    k = 0
    
    obj = 0
    for i in range(len(ta)):
        obj += ta[i] * np.sqrt( (td[i,0] - XX)**2 + (td[i,1] - YY)**2 )
    
    while ((k<imax)&(gap>tol)):
        k += 1
        temp = (ta/dd).sum()
        xnew = ((ta*td[:,0])/dd).sum()
        ynew = ((ta*td[:,1])/dd).sum()
        p1 = [xnew / temp, ynew / temp]
        dd = pdist(td,p1) + tol
#         newobj = round(sum(ta*dd), 3) 
# the 3 decimal points resoution as in R will make the function find the solution within 16 iterations.
        newobj = sum(ta*dd)
        gap = obj - newobj
        obj = newobj
    #print('the optimal location for the one facility is ', str(p1), ' and it took ', k, ' iterations to find the solution')
    return (p1, k)

mdata = pd.read_csv("swain.txt", sep="\t")

td = np.array(mdata[['x','y']])
ta = np.array(mdata['a'])
p1 = [0,0]
imax = 30
tol = 0.0000001

regionalWeber,k = weber(ta,td,p1,imax,tol)

points = [(mdata.loc[i, 'x'], mdata.loc[i, 'y']) for i in range(len(mdata))]#[:7]
spoints = [Point(x[0],x[1]) for x in points]

xs = [point.x for point in spoints]
ys = [point.y for point in spoints]


S = 10

sbuffers = [spoint.buffer(S) for spoint in spoints]

rings = [LineString(list(pol.exterior.coords)) for pol in sbuffers]
union = unary_union(rings)
faces = [geom for geom in polygonize(union)]
regionalWeber = Point(regionalWeber)
faceWebers = []
for face in faces:
    p1, _ = nearest_points(face, regionalWeber)
    p2, __ = nearest_points(face.boundary, regionalWeber)
    
    faceWebers.append(p1) if p1.distance(regionalWeber) < p2.distance(regionalWeber) else faceWebers.append(p2)
    
faceCoverage = []
faceTWD = []
x = []
y = []

for faceWeber in faceWebers:
    x.append(faceWeber.x)
    y.append(faceWeber.y)
    faceTWD.append((pdist(td, [faceWeber.x, faceWeber.y])*ta).sum())
    coverage = []
    for i, point in enumerate(spoints):
        if faceWeber.distance(point) <= S:
            coverage.append(i)
    else:
        faceCoverage.append(ta[coverage].sum()/ta.sum()*100)
        
source = pd.DataFrame({'x':x, 'y':y, 'COV':faceCoverage, 'TWD':faceTWD})
source['ID'] = range(len(source))
source.columns = ['X', 'Y', 'COV', 'TWD', 'ID']
sourceSorted = source.sort_values(["COV", "TWD"], ascending = (False, True)).reset_index()
sourceSorted.drop('index', axis=1,inplace=True)
sourceSorted.head()

n=len(faceWebers)
df = sourceSorted.groupby('COV').first().reset_index()[['ID', 'COV','TWD']]
Dominateds = np.zeros(n)
mask = np.ones(n, dtype=bool)
mask[df.ID] = False
Dominateds[mask] = 1
checked = np.ones(n)
i = 0
while checked.sum()!= 0:
    if Dominateds[i] == 0:
        row = df[df.ID == i]
        cov, twd = (row.COV.values[0], row.TWD.values[0])
        hi = df[(df.COV < cov) &  (df.TWD > twd)].ID
        Dominateds[hi] = 1
        checked[hi] = 0
    checked[i] = 0
    i += 1
source['NonDominated'] = np.array(1-Dominateds, dtype=bool)

source.NonDominated = np.array(source.NonDominated, dtype='int')
hmm = np.zeros(len(source))
hmm[source.loc[source.NonDominated == 1, 'ID']] = range(1,len(source[source.NonDominated == 1])+1)
source['color'] = pd.Series(hmm , dtype="category")
ND = source[source.NonDominated == 1]
D = source[source.NonDominated == 0]
brush = alt.selection(type='interval', resolve='global')

range_ = ['blue']

base = alt.Chart(D).mark_point(filled=True, color='lightgrey', size=10).encode(
